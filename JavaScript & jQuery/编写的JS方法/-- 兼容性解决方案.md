## 兼容性解决方案

```javascript
// 兼容IE8 不支持 node.getElementsByClassName
function elementsByClassName(node,classStr){
    var nodes = node.getElementsByTagName("*");
    var arry =[];
    for (var i = 0 ;i<nodes.length;i++){
        if(nodes[i].className === classStr){
            arry.push(nodes[i])
        }
    }
    return arry;
}
```
