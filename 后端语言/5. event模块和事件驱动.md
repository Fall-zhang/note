event 模块的使用

```js
// log.txt
厉害了，我的天
```

```js
// event.js
const events = require('events')
const fs = require('fs')
var eventEmitter = new events.EventEmitter();
fs.readFile('./log.txt',(err,data)=>{
  if(err){
    console.log('读取错误')
  }else{
    eventEmitter.emit('my_event',data);
  }
})
eventEmitter.on('my_event', (data) => {
  console.log('output:');
  console.log(data.toString())
});
```
